\begin{frame}[fragile]
  \frametitle{Haskell}
  \framesubtitle{Quicksort}

  \scriptsize
  \begin{code}
qsort :: Ord a => [a] -> [a]
qsort []     = []
qsort (x:xs) =
  qsort (filter (<= x) xs) ++ [x] ++ qsort (filter (> x) xs)
  \end{code}
  \normalsize
\end{frame}

sum (x:xs) = x + sum xs
